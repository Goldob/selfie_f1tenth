double check_sides(const sensor_msgs::LaserScan &ms, int observed_angles)
{
  using namespace std;

  double from_left = 0;
  double from_right = 0;
  deg_per_angle = ms.angle_increment * 180/PI;
  int max_angle_nr = (ms.angle_max - ms.angle_min)/ms.angle_increment;
  //  cout << "mr of angles: " << max_angle_nr<<endl;
  int center_angle_nr = max_angle_nr/2;
  observed_angles = observed_angles/deg_per_angle;
  //  cout << "observed_angles: " << observed_angles<<endl;
  int n = max_angle_nr - observed_angles;
  double min = 100;
  int min_nr = 0;
  double max = 0;
  int max_nr = 0;
  double sum = 0;
  double range_value = 0;
  double avg_range_value = 0;

  for(int i=n;  i<n+observed_angles;  i++)
  {
    range_value = ms.ranges[i];
    sum +=range_value;
    if(range_value < min)
    {
      min = range_value;
      min_nr = i;
    }
    if(range_value > max)
    {
      max = range_value;
      max_nr = i;
    }
  }
  from_right = sum/observed_angles;

  n = 1 + observed_angles;
   min = 100;
   min_nr = 0;
   max = 0;
   max_nr = 0;
   sum = 0;
   range_value = 0;
   avg_range_value = 0;

   for(int i=n;  i<n+observed_angles;  i++)
   {
     range_value = ms.ranges[i];
     sum +=range_value;
     if(range_value < min)
     {
       min = range_value;
       min_nr = i;
     }
     if(range_value > max)
     {
       max = range_value;
       max_nr = i;
     }
   }
   from_left = sum/observed_angles;

   return from_left - from_right;
}
